    public partial class Form1 : Form
    {
        public string connect = "Data Source=localhost\\SQLEXPRESS; database = exala2 ;Integrated Security=True;TrustServerCertificate=True";
        public Form1()
        {
            InitializeComponent();
        }

        private void button1_Click(object sender, EventArgs e)
        {
            string login = textBox1.Text;
            string password = textBox2.Text;
            SqlConnection con = new SqlConnection(connect);
            con.Open();
            string zapros = "SELECT * FROM users WHERE login=@login AND password=@password";
            SqlCommand comand = new SqlCommand(zapros, con);
            comand.Parameters.AddWithValue("@login", login);
            comand.Parameters.AddWithValue("password", password);
            SqlDataReader read = comand.ExecuteReader();
            if (login == "" || password == "")
            {
                MessageBox.Show("Пустые поля", "поля не заполнены", MessageBoxButtons.OK, MessageBoxIcon.Warning);

            }
            else
            {
                if (read.Read())
                {
                    MessageBox.Show("Добро пожаловать", "Пароль правильный", MessageBoxButtons.OK, MessageBoxIcon.Information);
                    Form2 f = new Form2();
                    f.Show();
                    this.Hide();


                }
                else
                {
                    MessageBox.Show("НЕ Добро пожаловать", "Не Пароль правильный", MessageBoxButtons.OK, MessageBoxIcon.Error);

                }
            }



        }

        private void button2_Click(object sender, EventArgs e)
        {
            Form2 f1 = new Form2();
            f1.Show();
            this.Hide();
        }
    }
